🔄 Function to extract tracks
def get_playlist_tracks(playlist_id, region, playlist_type):
    tracks_data = []
    
    results = sp.playlist_tracks(playlist_id)
    
    while results:
        for item in results['items']:
            track = item['track']
            
            if track:
                tracks_data.append({
                    "track_id": track['id'],
                    "track_name": track['name'],
                    "artist": track['artists'][0]['name'],
                    "region": region,
                    "playlist_type": playlist_type
                })
        
        if results['next']:
            results = sp.next(results)
        else:
            results = None
    
    return tracks_data
🔁 Extract all playlists

🧠 Code
def generate_positive_pairs(df):
    pairs = []
    
    grouped = df.groupby(["region", "playlist_type"])
    
    for _, group in grouped:
        track_ids = group["track_id"].tolist()
        
        for i in range(len(track_ids) - 1):
            pairs.append({
                "song_A": track_ids[i],
                "song_B": track_ids[i+1],
                "label": 1
            })
    
    return pd.DataFrame(pairs)
positive_pairs = generate_positive_pairs(df_tracks)

❌ STEP 4 — GENERATE NEGATIVE PAIRS
🎯 Logic:
song → random unrelated song
🧠 Code
import random

def generate_negative_pairs(df, n_samples):
    track_ids = df["track_id"].tolist()
    
    pairs = []
    
    for _ in range(n_samples):
        a = random.choice(track_ids)
        b = random.choice(track_ids)
        
        if a != b:
            pairs.append({
                "song_A": a,
                "song_B": b,
                "label": 0
            })
    
    return pd.DataFrame(pairs)
negative_pairs = generate_negative_pairs(df_tracks, len(positive_pairs))

def get_audio_features(track_ids):
    features = []
    
    for i in tqdm(range(0, len(track_ids), 100)):
        batch = track_ids[i:i+100]
        results = sp.audio_features(batch)
        
        for r in results:
            if r:
                features.append(r)
    
    return pd.DataFrame(features)
unique_tracks = df_tracks["track_id"].unique().tolist()
